### Example
```python
import requests

from lynker.lynker import Lynker

class Main:
    def __init__(self):
        self.lynker = Lynker()

        self.main()

    def main(self):
        r = requests.get("https://github.com/Neotoxic-off/lynk")

        print(self.lynker.feed(r.text))

if (__name__ == "__main__"):
    Main()

```
